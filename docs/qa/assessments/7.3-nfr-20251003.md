**Arrow Either Error Handling Pattern** (Comprehensive):

```kotlin
sealed class GeneratorError {
    data class ModuleNotFound(val moduleName: String) : GeneratorError()
    data class AggregateAlreadyExists(val name: String, val module: String) : GeneratorError()
    data class FileSystemError(val message: String, val cause: Throwable) : GeneratorError()
    data class TemplateError(val template: String, val cause: Throwable) : GeneratorError()
    data class UnsupportedType(val type: String, val supportedTypes: Set<String>) : GeneratorError()
    data class InvalidRegexPattern(val pattern: String, val error: String?) : GeneratorError()
    data class UnsafeRegexPattern(val pattern: String) : GeneratorError()
}

fun generateAggregate(...): Either<GeneratorError, AggregateInfo> = either {
    // Validation with early returns
    ensure(moduleExists) { GeneratorError.ModuleNotFound(moduleName) }
    ensure(!aggregateExists) { GeneratorError.AggregateAlreadyExists(name, module) }

    // Generation with error propagation
    val context = createContext(name, module, fields).bind()
    val files = renderTemplates(context).bind()
    writeFiles(files).bind()

    AggregateInfo(name, module, files)
}
```
