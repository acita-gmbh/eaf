#### Flowable JavaDelegate Pattern (from Story 6.2)

RunAnsiblePlaybookTask follows the same delegate pattern as DispatchAxonCommandTask:

```kotlin
@Component
@ConditionalOnProperty(
    prefix = "eaf.workflow",
    name = ["enabled"],
    havingValue = "true",
    matchIfMissing = true
)
class RunAnsiblePlaybookTask(
    private val ansibleExecutor: AnsibleExecutor,
    private val tenantContext: TenantContext,
    private val flowableMetrics: FlowableMetrics
) : JavaDelegate {
    override fun execute(execution: DelegateExecution) {
        val startTime = System.currentTimeMillis()

        // CRITICAL: Fail-closed tenant validation
        val currentTenant = tenantContext.getCurrentTenantId()
        val commandTenant = execution.getVariable("tenantId") as String
        require(currentTenant == commandTenant) {
            "Access denied: tenant context mismatch"
        }

        try {
            // Extract variables
            val playbookPath = execution.getVariable("playbookPath") as? String
                ?: throw BpmnError("MISSING_VARIABLE", "Required variable 'playbookPath' not found")

            val inventory = execution.getVariable("inventory") as? String
            val extraVars = execution.getVariable("extraVars") as? Map<String, Any> ?: emptyMap()

            // Execute Ansible playbook
            val result = ansibleExecutor.executePlaybook(
                playbookPath = playbookPath,
                inventory = inventory,
                extraVars = extraVars,
                tenantId = currentTenant
            )

            // Record metrics
            val duration = System.currentTimeMillis() - startTime
            flowableMetrics.recordProcessDuration(
                processInstanceId = execution.processInstanceId,
                durationMs = duration,
                processKey = execution.processDefinitionId
            )

            // Store result in process variables
            execution.setVariable("ansibleExitCode", result.exitCode)
            execution.setVariable("ansibleStdout", result.stdout)

            logger.info("Ansible playbook executed successfully: $playbookPath (${duration}ms)")

        } catch (ex: AnsibleExecutionException) {
            // Record error metric
            flowableMetrics.recordBpmnError(ex.errorCode, execution.processDefinitionId)

            // Throw BpmnError for error boundary event routing (enables Story 6.5 compensation)
            throw BpmnError(
                ex.errorCode,
                "Ansible execution failed: ${ex.message}",
                execution.id
            )
        }
    }
}
```

#### Ansible Execution via SSH

The Dockets Analysis identifies SSH as the connection method for Ansible execution. Using JSch library (Java SSH client):

```kotlin
@Component
class AnsibleExecutor(
    private val environment: Environment
) {

    fun executePlaybook(
        playbookPath: String,
        inventory: String?,
        extraVars: Map<String, Any>,
        tenantId: String
    ): AnsibleResult {
        val sshHost = environment.getProperty("eaf.ansible.ssh.host") ?: "localhost"
        val sshPort = environment.getProperty("eaf.ansible.ssh.port")?.toInt() ?: 22
        val sshUser = environment.getProperty("eaf.ansible.ssh.username") ?: "ansible"
        val sshKey = environment.getProperty("eaf.ansible.ssh.private-key-path")

        val session = JSch().getSession(sshUser, sshHost, sshPort)

        // Configure SSH authentication
        if (sshKey != null) {
            JSch().addIdentity(sshKey)
        }
        session.setConfig("StrictHostKeyChecking", "no") // Dev only - production should validate
        session.setTimeout(300_000) // 5 minute timeout

        try {
            session.connect()

            // Build ansible-playbook command
            val extraVarsJson = ObjectMapper().writeValueAsString(extraVars)
            val inventoryArg = if (inventory != null) "-i $inventory" else ""
            val command = """
                ansible-playbook $inventoryArg \
                  --extra-vars '$extraVarsJson' \
                  $playbookPath
            """.trimIndent()

            // Execute command via SSH
            val channel = session.openChannel("exec") as ChannelExec
            channel.setCommand(command)

            val stdout = ByteArrayOutputStream()
            val stderr = ByteArrayOutputStream()
            channel.outputStream = stdout
            channel.setErrStream(stderr)

            channel.connect()

            // Wait for command completion
            while (!channel.isClosed) {
                Thread.sleep(100)
            }

            val exitCode = channel.exitStatus
            channel.disconnect()

            logger.info("Ansible playbook executed: $playbookPath (exit code: $exitCode)")

            if (exitCode != 0) {
                throw AnsibleExecutionException(
                    errorCode = "ANSIBLE_FAILED",
                    message = "Ansible playbook failed with exit code $exitCode",
                    stdout = stdout.toString(),
                    stderr = stderr.toString()
                )
            }

            return AnsibleResult(
                exitCode = exitCode,
                stdout = stdout.toString(),
                stderr = stderr.toString()
            )

        } finally {
            session.disconnect()
        }
    }
}

data class AnsibleResult(
    val exitCode: Int,
    val stdout: String,
    val stderr: String
)

class AnsibleExecutionException(
    val errorCode: String,
    message: String,
    val stdout: String,
    val stderr: String
) : RuntimeException(message)
```

#### Flowable Metrics Implementation (NEW - Addresses PO Validation Gaps)

```kotlin
@Component
class FlowableMetrics(
    private val meterRegistry: MeterRegistry,
    private val processEngine: ProcessEngine
) {

    // METRIC 1: Active process instances (by process definition)
    @Scheduled(fixedRate = 30000) // Every 30s
    fun recordProcessInstanceMetrics() {
        val runtime = processEngine.runtimeService

        runtime.createProcessInstanceQuery()
            .list()
            .groupBy { it.processDefinitionKey }
            .forEach { (key, instances) ->
                Gauge.builder("flowable.process.instances.active", instances) { it.size.toDouble() }
                    .tag("process_key", key)
                    .register(meterRegistry)
            }

        // Suspended instances (potential issues)
        val suspended = runtime.createProcessInstanceQuery().suspended().count()
        meterRegistry.gauge("flowable.process.instances.suspended", suspended.toDouble())
    }

    // METRIC 2: Process execution duration
    fun recordProcessDuration(processInstanceId: String, durationMs: Long, processKey: String) {
        Timer.builder("flowable.process.duration")
            .tag("process_key", processKey)
            .tag("status", "completed")
            .register(meterRegistry)
            .record(durationMs, TimeUnit.MILLISECONDS)
    }

    // METRIC 3: BPMN error rates (for rollback trigger monitoring)
    fun recordBpmnError(errorCode: String, processKey: String) {
        Counter.builder("flowable.process.errors")
            .tag("error_code", errorCode)
            .tag("process_key", processKey)
            .register(meterRegistry)
            .increment()
    }

    // METRIC 4: Signal delivery success rate
    fun recordSignalDelivery(signalName: String, delivered: Boolean) {
        Counter.builder("flowable.signal.deliveries")
            .tag("signal_name", signalName)
            .tag("status", if (delivered) "success" else "failed")
            .register(meterRegistry)
            .increment()
    }

    // METRIC 5: Dead letter queue depth
    @Scheduled(fixedRate = 60000) // Every 60s
    fun recordDeadLetterQueue() {
        val management = processEngine.managementService
        val deadLetterJobs = management.createDeadLetterJobQuery().count()
        meterRegistry.gauge("flowable.jobs.dead_letter", deadLetterJobs.toDouble())

        if (deadLetterJobs > 100) {
            logger.error("Dead letter queue depth critical: $deadLetterJobs jobs")
        }
    }
}
```

**Rollback Procedure**:
1. Set `eaf.workflow.enabled=false` in application.yml (feature flag - disables all Flowable beans)
2. Restart application (Flowable engine will not initialize)
3. Investigate root cause using Flowable metrics and logs
4. Apply fix or restore from schema backup
5. Re-enable: `eaf.workflow.enabled=true` and restart

#### Backup Procedures

**Flowable Schema Backup Script** (`scripts/backup-flowable-schema.sh`):

```bash
#!/bin/bash
# Flowable Schema Backup for PostgreSQL
# Execute BEFORE any Flowable version upgrade or schema migration

BACKUP_DIR="./backups/flowable"
TIMESTAMP=$(date +%Y%m%d_%H%M%S)
BACKUP_FILE="${BACKUP_DIR}/flowable_schema_${TIMESTAMP}.dump"

mkdir -p "$BACKUP_DIR"

# Backup Flowable schema only (not entire database)
pg_dump -h ${POSTGRES_HOST:-localhost} \
        -U ${POSTGRES_USER:-postgres} \
        -d ${POSTGRES_DB:-eaf_database} \
        --schema=flowable \
        --format=custom \
        --file="$BACKUP_FILE"

if [ $? -eq 0 ]; then
    echo "✅ Flowable schema backed up: $BACKUP_FILE"

    # Verify backup integrity
    pg_restore --list "$BACKUP_FILE" > /dev/null

    if [ $? -eq 0 ]; then
        echo "✅ Backup verified successfully"
    else
        echo "❌ Backup verification failed!"
        exit 1
    fi
else
    echo "❌ Backup failed!"
    exit 1
fi

# Cleanup old backups (keep last 7 days)
find "$BACKUP_DIR" -name "flowable_schema_*.dump" -mtime +7 -delete
echo "🗑️ Cleaned up backups older than 7 days"
```

```bash
#!/bin/bash
# Flowable Schema Restore

           -U ${POSTGRES_USER:-postgres} \
           -d ${POSTGRES_DB:-eaf_database} \

if [ $? -eq 0 ]; then
    echo "✅ Flowable schema restored from: $BACKUP_FILE"
    echo "🚀 Start application to validate"
else
    echo "❌ Restore failed!"
    exit 1
fi
```

### Top Issues

**MEDIUM-01: Gauge Registration Memory Leak**
- **Severity**: MEDIUM
- **Location**: `FlowableMetrics.kt:73-78, 81-84, 87-91`
- **Finding**: Gauge metrics are re-registered on every `@Scheduled` invocation (every 30s/60s), creating duplicate metric instances in MeterRegistry. This causes gradual memory accumulation.
- **Impact**: Production memory leak - approximately 3-5 new Gauge instances per minute → ~4,320 instances per day → potential OOM after several days of uptime
- **Suggested Action**: Refactor to register Gauges ONCE during initialization and observe mutable state (use `AtomicLong` or similar for state management)
- **Suggested Owner**: dev
- **Reference**: [Micrometer Gauge Best Practices](https://micrometer.io/docs/concepts#_gauges)

**Pattern Example** (correct):
```kotlin
@Component
class FlowableMetrics {
    private val activeInstancesByProcessKey = ConcurrentHashMap<String, AtomicLong>()

    init {
        // Register gauges ONCE during component initialization
        // Gauges observe state via lambda that reads current values
    }

    @Scheduled(fixedRate = 30000)
    fun recordProcessInstanceMetrics() {
        // UPDATE state that gauges observe (don't re-register)
        val counts = runtime.createProcessInstanceQuery()...
        activeInstancesByProcessKey.clear()
        counts.forEach { activeInstancesByProcessKey[it.key] = AtomicLong(it.value) }
    }
}
```
